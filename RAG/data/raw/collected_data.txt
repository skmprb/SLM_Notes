# In-memory session storage (use Redis/DB in production)
sessions: Dict[str, List[Dict]] = {}

class DynamicSessionManager:
    def __init__(self, bucket: str, folder_name: str, region: str = "us-east-1"):
        self.bucket = bucket
        self.folder_name = folder_name.rstrip('/')
        self.s3 = boto3.client('s3', region_name=region)

    def list_sessions(self) -> List[Dict[str, Any]]:
        sessions = []
        try:
            paginator = self.s3.get_paginator('list_objects_v2')
            for page in paginator.paginate(Bucket=self.bucket, Prefix=f"{self.folder_name}/"):
                for obj in page.get('Contents', []):
                    if '/session.json' in obj['Key']:
                        session_id = self._extract_session_id(obj['Key'])
                        if session_id:
                            session_data = self._read_json(obj['Key'])
                            if session_data:
                                session_data['session_id'] = session_id
                                sessions.append(session_data)
        except ClientError:
            pass
        return sessions

    def get_session(self, session_id: str) -> Optional[Dict[str, Any]]:
        session_key = f"{self.folder_name}//session_{session_id}/session.json"
        session_data = self._read_json(session_key)
        if session_data:
            session_data['session_id'] = session_id
        return session_data

    def get_session_messages(self, session_id: str) -> List[Dict[str, Any]]:
        messages = []
        messages_prefix = f"{self.folder_name}//session_{session_id}/agents/agent_default/messages/"
        try:
            paginator = self.s3.get_paginator('list_objects_v2')
            for page in paginator.paginate(Bucket=self.bucket, Prefix=messages_prefix):
                for obj in page.get('Contents', []):
                    if obj['Key'].endswith('.json'):
                        message_data = self._read_json(obj['Key'])
                        if message_data:
                            filename = obj['Key'].split('/')[-1]
                            if filename.startswith('message_'):
                                index = int(filename.replace('message_', '').replace('.json', ''))
                                message_data['message_index'] = index
                                messages.append(message_data)
            messages.sort(key=lambda x: x.get('message_index', 0))
        except (ClientError, ValueError):
            pass
        return messages

    def get_session_details(self, session_id: str) -> Optional[Dict[str, Any]]:
        session_data = self.get_session(session_id)
        if session_data:
            session_data['messages'] = self.get_session_messages(session_id)
            session_data['message_count'] = len(session_data['messages'])
        return session_data

    def _extract_session_id(self, s3_key: str) -> Optional[str]:
        parts = s3_key.split('/')
        for part in parts:
            if part.startswith('session_'):
                return part.replace('session_', '')
        return None

    def _read_json(self, key: str) -> Optional[Dict[str, Any]]:
        try:
            response = self.s3.get_object(Bucket=self.bucket, Key=key)
            return json.loads(response['Body'].read().decode('utf-8'))
        except (ClientError, json.JSONDecodeError):
            return None

S3_BUCKET = os.getenv("S3_BUCKET", "qbddassistant-strands")

# Session manager will be created per request with agent-specific folder

class ChatRequest(BaseModel):
    prompt: str
    session_id: Optional[str] = None
    agent_id: Optional[str] = None

class SessionResponse(BaseModel):
    session_id: str
    messages: List[Dict]

def invoke_agent(agent_id: str, payload: dict, session_id: str):
    client = boto3.client(
        "bedrock-agentcore", 
        region_name="us-east-1",
        config=boto3.session.Config(read_timeout=60)
    )
    response = client.invoke_agent_runtime(
        agentRuntimeArn=f"arn:aws:bedrock-agentcore:us-east-1:599149960404:runtime/{agent_id}",
        runtimeSessionId=session_id or "12345678-1234-5678-9abc-123456789012",
        payload=json.dumps(payload)
    )
    return response

@router.post("/chat/new")
async def create_session():
    session_id = str(uuid.uuid4())
    sessions[session_id] = []
    return {"session_id": session_id}

@router.get("/chat/{session_id}")
async def get_session(session_id: str):
    if session_id not in sessions:
        raise HTTPException(status_code=404, detail="Session not found")
    return SessionResponse(session_id=session_id, messages=sessions[session_id])

@router.delete("/chat/{session_id}")
async def clear_session(session_id: str):
    if session_id not in sessions:
        raise HTTPException(status_code=404, detail="Session not found")
    sessions[session_id] = []
    return {"message": "Session cleared"}

@router.get("/sessions")
async def list_all_sessions(agent_id: str):
    """List all sessions from S3 for specific agent"""
    agent_data = db_client.get_agent_by_id(agent_id)
    if not agent_data:
        raise HTTPException(status_code=404, detail=f"Agent {agent_id} not found")
    
    app_name = agent_data.get('agentProperties', {}).get('app') or agent_data.get('displayName', 'production')
    session_manager = DynamicSessionManager(S3_BUCKET, app_name)
    
    sessions_data = session_manager.list_sessions()
    return {"sessions": sessions_data, "count": len(sessions_data)}

                            break

                            yield f": keepalive\n\n"

        finally:
            print("🏁 Stream ending")
            yield f"data: {json.dumps({'end': True})}\n\n"

# In-memory session storage (use Redis/DB in production)

class DynamicSessionManager:
    def __init__(self, bucket: str, folder_name: str, region: str = "us-east-1"):
        self.bucket = bucket
        self.folder_name = folder_name.rstrip('/')
        self.s3 = boto3.client('s3', region_name=region)

    def list_sessions(self) -> List[Dict[str, Any]]:
        sessions = []
        try:
            paginator = self.s3.get_paginator('list_objects_v2')
            for page in paginator.paginate(Bucket=self.bucket, Prefix=f"{self.folder_name}/"):
                for obj in page.get('Contents', []):
                    if '/session.json' in obj['Key']:
                        session_id = self._extract_session_id(obj['Key'])
                        if session_id:
                            session_data = self._read_json(obj['Key'])
                            if session_data:
                                session_data['session_id'] = session_id
                                sessions.append(session_data)
        except ClientError:
            pass
        return sessions

    def get_session(self, session_id: str) -> Optional[Dict[str, Any]]:
        session_key = f"{self.folder_name}//session_{session_id}/session.json"
        session_data = self._read_json(session_key)
        if session_data:
            session_data['session_id'] = session_id
        return session_data

    def get_session_messages(self, session_id: str) -> List[Dict[str, Any]]:
        messages = []
        messages_prefix = f"{self.folder_name}//session_{session_id}/agents/agent_default/messages/"
        try:
            paginator = self.s3.get_paginator('list_objects_v2')
            for page in paginator.paginate(Bucket=self.bucket, Prefix=messages_prefix):
                for obj in page.get('Contents', []):
                    if obj['Key'].endswith('.json'):
                        message_data = self._read_json(obj['Key'])
                        if message_data:
                            filename = obj['Key'].split('/')[-1]
                            if filename.startswith('message_'):
                                index = int(filename.replace('message_', '').replace('.json', ''))
                                message_data['message_index'] = index
                                messages.append(message_data)
            messages.sort(key=lambda x: x.get('message_index', 0))
        except (ClientError, ValueError):
            pass
        return messages

    def get_session_details(self, session_id: str) -> Optional[Dict[str, Any]]:
        session_data = self.get_session(session_id)
        if session_data:
            session_data['messages'] = self.get_session_messages(session_id)
            session_data['message_count'] = len(session_data['messages'])
        return session_data

    def _extract_session_id(self, s3_key: str) -> Optional[str]:
        parts = s3_key.split('/')
        for part in parts:
            if part.startswith('session_'):
                return part.replace('session_', '')
        return None

    def _read_json(self, key: str) -> Optional[Dict[str, Any]]:
        try:
            response = self.s3.get_object(Bucket=self.bucket, Key=key)
            return json.loads(response['Body'].read().decode('utf-8'))
        except (ClientError, json.JSONDecodeError):
            return None

                    events = response.get("events", [])
                    new_token = response.get("nextForwardToken")
                    
                    if events:
                        no_data_count = 0
                        for event in events:
                            clean_message = re.sub(r"\x1b\[[0-9;]*m", "", event["message"].strip())
                            log_data = {"timestamp": event["timestamp"], "message": clean_message}
                            yield f"data: {json.dumps(log_data)}\n\n"
                    else:
                        no_data_count += 1
                        if no_data_count > 10:  # No data for 20 seconds, end stream
                            break
                    
                    if new_token == next_token or new_token in seen_tokens:
                        if no_data_count > 5:
                            yield f": keepalive\n\n"
                        await asyncio.sleep(2)
                    else:
                        if next_token:
                            seen_tokens.add(next_token)
                        next_token = new_token
                        await asyncio.sleep(0.3)
                        
                except Exception as e:
                    print(f"❌ Error polling logs: {str(e)}")
                    await asyncio.sleep(1)
        
        except Exception as e:
            print(f"❌ Fatal error: {str(e)}")
            yield f"data: {json.dumps({'error': f'CloudWatch error: {str(e)}'})}\n\n"
        finally:
            print("🏁 Stream ending")
            yield f"data: {json.dumps({'end': True})}\n\n"
    
    return StreamingResponse(
        log_stream(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache, no-store, must-revalidate",
            "Connection": "keep-alive",
            "X-Accel-Buffering": "no",
            "Pragma": "no-cache",
            "Expires": "0",
            "Access-Control-Allow-Origin": "*"
        }
    )
